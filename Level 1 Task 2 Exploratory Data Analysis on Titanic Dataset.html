clean = df.copy()

clean["survived"] = clean["survived"].astype(int)
clean["pclass"] = pd.Categorical(clean["pclass"], categories=[1,2,3], ordered=True)
clean["sex"] = clean["sex"].astype("category")
if "embarked" in clean.columns:
    clean["embarked"] = clean["embarked"].astype("category")

# Age: fill missing with median by (pclass, sex), fallback to global median
if set(["pclass","sex","age"]).issubset(clean.columns):
    med = clean.groupby(["pclass","sex"])["age"].transform("median")
    clean["age"] = clean["age"].fillna(med)
clean["age"] = clean["age"].fillna(clean["age"].median())

# Embarked: mode
if "embarked" in clean.columns and clean["embarked"].isna().any():
    clean["embarked"] = clean["embarked"].fillna(clean["embarked"].mode()[0])

# Fare: median by class
if "fare" in clean.columns and clean["fare"].isna().any():
    clean["fare"] = clean.groupby("pclass")["fare"].transform(lambda s: s.fillna(s.median()))
    clean["fare"] = clean["fare"].fillna(clean["fare"].median())

# Drop very-high-missing columns (like cabin)
for col in ["cabin","deck"]:
    if col in clean.columns and clean[col].isna().mean() > 0.65:
        clean = clean.drop(columns=[col])

# Features
clean["family_size"] = clean.get("sibsp",0) + clean.get("parch",0) + 1
clean["is_alone"] = (clean["family_size"] == 1).astype(int)

print(clean.isna().sum())
